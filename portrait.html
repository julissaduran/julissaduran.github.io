Creating a self portait using P5js


Code: function setup() {
createCanvas(600, 600); }
function draw() {
background(255); }
function draw() {
noStroke ()
fill(51,0,0) //hair curved
arc(260, 260, 300, 380, QUARTER_PI + HALF_PI, QUARTER_PI, OPEN);
noStroke()
fill(247, 221, 212) //head
ellipse(260, 260, 220, 280);
noStroke()
fill(51, 0, 0); //Bang
arc(300, 150, 150, 100, 0, PI + QUARTER_PI, CHORD);
arc(180, 120, 140, 148, 0, 0.75 * Math.PI);
stroke(21, 19, 19); //eyebrows
strokeWeight(5);
noFill();
arc(210, 235, 45, 15, PI, TWO_PI, OPEN);
arc(310, 235, 45, 15, PI, TWO_PI, OPEN);
strokeWeight(10) //eyes
fill(32, 19, 19);
ellipse(210, 260, 28, 25);
ellipse(310, 260, 28, 25);
noStroke() //eyeballs
fill(255)
ellipse(207, 252, 8, 8);
ellipse(306, 252, 8, 8);
fill(255, 255, 153) //nose
triangle(260, 270, 270, 300, 250, 300);
fill(255, 102, 102); //mouth
arc(260, 320, 60, 60, 0, PI);
rectMode(CENTER) //hoodie
fill(0, 0, 0)
rect(260, 470, 205, 150, 20);
fill(255, 204, 229)
fill(255)
ellipse(300, 432, 15, 15);
ellipse(220, 432, 15, 15);
rect(300, 460, 8, 65, 20);
rect(220, 460, 8, 65, 20); //Neck
fill(247, 221, 212)
rect(260, 380, 50, 60, 10);
}
